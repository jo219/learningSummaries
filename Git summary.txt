git config --global color.ui auto


git init .

# to staging area ("git status" for check)
git add <file name>
git rm --cached <file name>

# to only current dir downwards
git add .
git rm -r --cached . 


# get to commit area
git commit -m "init"

git log
git show <HASH>
git diff
git restore # cancel what on diff

# change message of last commit
git commit --amend --m  "add something"

git branch  		# to show (local)
git branch -M main  # change name
git branch -r 		# list remote only
		   -a 		# list all

git branch   <New Branch Name>
git checkout <New Branch Name>
git checkout - b <New Branch Name>  # 2 steps
git checkout -						# to previous branch
git branch -d <Branch To Delete>
